  # pivot and pos is important #
  def Qsort(lt, rt)
      if lt < rt: # most important
          pivot = a[rt]
          pos = lt
          for i in range(lt, rt): # does not include *rt*
              if a[i] < pivot:
                  a[i], a[pos] = a[pos], a[i] # python list exchange
